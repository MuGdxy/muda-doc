87 MUDA_GENERIC static bool allow_ignore(const SpatialPartitionCell& l,
88 const SpatialPartitionCell& r)
89 {
90 if(l.is_phantom() && r.is_phantom())

92 return true;

94
95 const SpatialPartitionCell* arr[] = {&l, &r};
96
97 U32 pass = l.ctlbit.pass;
98 U32 common_overlap = l.ctlbit.overlap & r.ctlbit.overlap;
99#pragma unroll
100 for(U32 i = 0; i < 2; ++i)
101 {
102 U32 encode_home = (1 << arr[i]->ctlbit.home);
103 if(arr[i]->ctlbit.home < pass && (common_overlap & encode_home))
104 {
105 return true;
106 }
107 }
108 return false;
109 }
